// 1픽셀 = 1m //
// 기본 맵 사이즈 100, 100 //
// 중심 50,50 //
// my_center, my_center 를 확대 시킨창을 따로 보여준다. //

#include "ros/ros.h"
#include "sensor_msgs/LaserScan.h"
#include <math.h>
#include <opencv2/opencv.hpp>
#include "tf/transform_listener.h"
#include "project_ros_cpp/get_degree.h"
#define _USE_MATH_DEFINES
#include <cstdio>
#define RAD2DEG(x) ((x)*180.0/M_PI)

using namespace cv;
using namespace std;

// 초기 맵 세팅 //
// 중심 (50, 50) //
// 창1 : 전체 맵 사이즈 100x100 //
// 창2 : (my_center, my_center)를 중심으로 가지는 20x20 매트릭스 //
// + 창2 : 확대 30배, 1m = 30px, #define enlarge 30 //
// + 창2 : 확대 후, 상하좌우 검은색 패딩 10, #define padding 0 //
// padding은 확대 후 매트릭스의 여백 설정 ( 10 입력하면 상하좌우 10씩 늘어난다. )
// 1픽셀 = 1m, 30배 확대햇으면 30픽셀 = 1m 이다.

#define enlarge 30
#define padding 0

Mat my_map(100, 100, CV_32FC3, Scalar(0));
// 100x100, float32형 3채널, 0= 검은색으로 채운다.
Mat enlarge_my_map; // 확대될 mat
Mat temp_my_map; // 좌표축 그리기 전에 원본 맵을 보존 시킨다.

// 초기 맵에 그릴 초기 좌표, 좌표계
Point2d my_center(my_map.size().width/2, my_map.size().height/2);
Point2d carto_center;
Point2d my_axis_x(my_map.size().width/8,0);
Point2d my_axis_y(0,my_map.size().height/8);
Point2d my_axis_x_rotation, my_axis_y_rotation;
// 좌표축 갱신을 위한 선언
int re_axis = 0;

int m_a_count = 0; // 유효한 거리값 개수 확인을 위한 선언
Point2d detection; // 유효한 좌표를 위한 변수 선언

// 카토그래퍼에서 좌표를 받기 위한 함수 //
void from_cartographer_get_point()
{
	tf::TransformListener tfl;
	tf::StampedTransform for_get_coordinate;
	tf::Vector3 Coordinate_3D;

	tfl.waitForTransform("/odom", "/base_footprint", ros::Time(), ros::Duration(1.0));
	tfl.lookupTransform("/odom", "/base_footprint", ros::Time(), for_get_coordinate);
	Coordinate_3D = for_get_coordinate.getOrigin();

	//std::cout<<"시작 점으로 부터 현재 위치 [metric] (x, y, z)"<<std::endl;
	//std::cout<<"("<<Coordinate_3D.getX()<<", "<<Coordinate_3D.getY()<<", "<<Coordinate_3D.getZ()<<")"<<std::endl;

	// 이동한 만큼 센터를 수정
	carto_center.x = my_center.x + Coordinate_3D.getX();
	carto_center.y = my_center.y + Coordinate_3D.getY();
	printf("(x, y) : [%.3f, %.3f]\n", carto_center.x , carto_center.y);
}



// IMU에서 각도를 받아 좌표축을 수정하는 함수 //
float roll = 0.0;
float pitch = 0.0;
float yaw = 0.0;
void get_degree_from_imu(const project_ros_cpp::get_degree::ConstPtr& msg)
{
	// msg에서 data를 꺼낸다.
	roll = msg->roll;
	pitch = msg->pitch;
	yaw = msg->yaw;
	printf("get degree : %f, %f, %f", roll, pitch, yaw);

}
// 포인트 회전을 위한 함수 
Point2d point_rotation(Point2d a, double b)
{
	Point2d temp_a(a.x,a.y);

	temp_a.x = a.x*cos(b) - a.y*sin(b);
	temp_a.y = a.x*sin(b) + a.y*cos(b);

	std::cout<<temp_a.x<<" "<<temp_a.y<< std::endl;
	return temp_a;
}
void from_IMU_modify_axis()
{
	// Subscriber 선언
	// degree_publisher.py에서 지정한 Topic명 project_degree로부터
	// 1개의 큐를 msgCallback으로 보낸다.
	ros::NodeHandle nh;
	ros::Subscriber imu_sub = nh.subscribe("project_degree", 1, get_degree_from_imu);

	my_axis_x_rotation = point_rotation(my_axis_x,-yaw) + my_center;
	my_axis_y_rotation = point_rotation(my_axis_y,yaw);
	my_axis_y_rotation.y = -my_axis_y_rotation.y;
	my_axis_y_rotation = my_axis_y_rotation+my_center;
	line(temp_my_map, carto_center, my_axis_x_rotation, Scalar(0, 0, 255), 2, LINE_AA); // x 빨간색
	line(temp_my_map, carto_center, my_axis_y_rotation, Scalar(0, 255, 0), 2, LINE_AA); // y 초록색
	re_axis = 0;
}















void scanCallback(const sensor_msgs::LaserScan::ConstPtr& scan)
{
	printf("here? Tag0-1\n");
	from_cartographer_get_point(); // 함수를 호출해서 my_center를 수정한다.
	printf("here? Tag0-2\n");

	// 맵 데이터 획득부분//
	// ydlidar로부터 직접 획득한다. //
	// 전체 데이터 갯수를 획득한다. //
	int count = scan->scan_time / scan->time_increment;

	// count값의 최대치를 직접 ydlidar를 켜서 실험을 통해 구한다.
	// m_a_count 변수의 값을 보고 숫자를 조정한다.
	// 매번 채워져있는 수가 변하므로 함수가 종료되면 삭제되었다가 다시 선언되는것이 좋다.
	double my_array[1000][2];
	printf("here? Tag1\n");

	// 전체 데이터중 유효한 데이터를 찾는 함수 //
	m_a_count = 0;
	for(int i = 0; i < count; i++)
	{
		float degree = RAD2DEG(scan->angle_min + scan->angle_increment * i);

		if( degree<180 && degree>-180 )
		{
			if( isinf(scan->ranges[i]) || scan->ranges[i]<0.1 || scan->ranges[i]>10 )
			{
			// ydlidar의 datasheet 참조 //
			// 최소 0.1m, 최대 10m의 측정을 보장한다. //
			// 무한대, 0.1m미만, 10m초과 값을 버린다. //
			}
			else
			{
				my_array[m_a_count][0] = degree;
				my_array[m_a_count][1] = scan->ranges[i];

				// openCV로 line을 생성하기 위한 좌표를 계산한다.
				detection.x =  carto_center.x + my_array[i][1]*cos(my_array[i][0]*M_PI/180.0);
				detection.y =  carto_center.y + -my_array[i][1]*sin(my_array[i][0]*M_PI/180.0);

				// my_map의 carto_center로부터 detection까지 흰색으로(255), 굵기 2, 라인 안티앨리어싱
				line(my_map, carto_center, detection, Scalar(255,255,255), 2, LINE_AA);
				// my_map의 detection위치에 반지름 0.5의 검은색 원
				circle(my_map, detection, 0.5, Scalar(0), 1);
				
				// 어디에 그렸는지 출력
				//printf("angle, distance : [%.3f, %.3f]\n", degree, scan->ranges[i]);
				++m_a_count; 
			}
		}
	}

	// 원본 맵을 보존한다.
	printf("here? Tag2\n");
	temp_my_map = my_map.clone();
	printf("here? Tag3\n");

	// 좌표축은 5번 마다 갱신 //
	if(re_axis<5)
	{
		// temp_my_map의 my_center부터 my_axis_x_rotation까지 흰색으로(255), 굵기 2, 라인 안티앨리어싱
		line(temp_my_map, carto_center, my_axis_x_rotation, Scalar(0, 0, 255), 2, LINE_AA); // x 빨간색
		line(temp_my_map, carto_center, my_axis_y_rotation, Scalar(0, 255, 0), 2, LINE_AA); // y 초록색
	}
	else
	{
		//from_IMU_modify_axis();
	}

	moveWindow("my_map_window", 100, 100); // 모니터의 좌측 상단이 원점이다. (+x -y)
	namedWindow("my_map_window", WINDOW_AUTOSIZE); // WINDOW_AUTOSIZE = 할당된 행렬에 딱 맞게 크기를 생성함, 크기조절 불가
	imshow("my_map_window", temp_my_map);


	// 원본 200 x 200
	// 100,100을 중심으로
	// 20x20 영역을 30배 확대한다.
	// Rect으로 참조영역을 만든다.
	// x,y,w,h
	Rect reference((int)(carto_center.y-10), (int)(carto_center.x-10), 20, 20);
	Mat for_range_map = temp_my_map(reference);
	for_range_map = for_range_map.clone();
	cout <<  for_range_map.size() << endl;
	cout <<  for_range_map.cols << endl;
	cout <<  for_range_map.rows << endl;
	// for_range_map를 확대해서 enlarge_my_map으로 복사
	cv::resize( for_range_map, enlarge_my_map, Size(for_range_map.cols*enlarge,for_range_map.rows*enlarge), 0, 0, INTER_NEAREST);
	moveWindow("my_map_enlarge_window", 300, 100);
	namedWindow("my_map_enlarge_window", WINDOW_AUTOSIZE); // WINDOW_AUTOSIZE = 할당된 행렬에 딱 맞게 크기를 생성함, 크기조절 불가
	imshow("my_map_enlarge_window", enlarge_my_map);
	

	waitKey(1);
	printf("axis draw count : %d\n", re_axis);
	++re_axis;


}

int main(int argc, char **argv)
{
	// ROS node 통신을 위한 선언 //
	ros::init(argc, argv, "draw_map_node");
	ros::NodeHandle n;

	// 초기 좌표축 x,y 설정 //
	my_axis_x_rotation = my_axis_x + my_center;
	my_axis_y_rotation = my_axis_y;
	my_axis_y_rotation.y = -my_axis_y_rotation.y;
	my_axis_y_rotation = my_axis_y_rotation + my_center;

	// 카토그래퍼로 보내지는 /scan 토픽을 읽어서 scanCallback함수를 실행한다.
	ros::Subscriber ydlidar_sub = n.subscribe<sensor_msgs::LaserScan>("/scan", 1000, scanCallback);
	ros::spin();

	return 0;
}
